## 任务组 A：预览性能优化（MJPEG 流）

**当前问题：** 前端 `setInterval(updatePreview, 50)` 每 50ms 轮询 `/api/frame`，20 次/秒，无论是否有新帧。每次请求建立/断开 HTTP 连接，开销大。

**方案：** 新增 `/api/stream` 端点返回 `multipart/x-mixed-replace` MJPEG 流，浏览器单连接持续接收。前端 `<img src="/api/stream">` 即可，无需 JS 轮询。保留 `/api/frame` 作为单帧快照向后兼容。

```python
@app.route('/api/stream')
def video_stream():
    """MJPEG 实时流：单连接持续推送，仅在 frame_id 变化时发送新帧。"""
    boundary = 'frame'

    def generate():
        last_id = -1
        while True:
            frame = latest_frame
            fid = frame_id
            if frame is not None and fid != last_id:
                last_id = fid
                yield (
                    b'--' + boundary.encode() + b'\r\n'
                    b'Content-Type: image/jpeg\r\n\r\n' + frame + b'\r\n'
                )
            # 上限 ~15 FPS，避免空转占满 CPU
            time.sleep(1.0 / 15)

    return Response(
        generate(),
        mimetype=f'multipart/x-mixed-replace; boundary={boundary}',
        headers={'Cache-Control': 'no-cache, private'},
    )
```

```python
"""状态持久化：将非瞬态状态序列化为 JSON 文件。

仅持久化字段：saved_photos、flash_enabled、esp32_cam_ip、esp32_cam_ip_time。
不持久化 latest_frame、is_capturing 等瞬态字段。
"""
import json
import os
import threading

_lock = threading.Lock()

DEFAULT_STATE = {
    'saved_photos': [],
    'flash_enabled': False,
    'esp32_cam_ip': None,
    'esp32_cam_ip_time': 0,
}


def load_state(path):
    """加载状态文件。文件不存在或损坏时返回默认状态。"""
    if not os.path.exists(path):
        return dict(DEFAULT_STATE)
    try:
        with open(path, 'r', encoding='utf-8') as f:
            data = json.load(f)
    except (json.JSONDecodeError, OSError):
        return dict(DEFAULT_STATE)
    # 合并默认键，防止旧文件缺字段
    merged = dict(DEFAULT_STATE)
    merged.update(data)
    return merged


def save_state(path, state):
    """原子写入状态文件（临时文件 + os.replace）。"""
    with _lock:
        tmp = path + '.tmp'
        with open(tmp, 'w', encoding='utf-8') as f:
            json.dump(state, f, ensure_ascii=False, indent=2)
        os.replace(tmp, path)
```

```python
    global saved_photos, flash_enabled, esp32_cam_ip, esp32_cam_ip_time
```

## 任务组 C：相册与上传目录分离

**当前问题：** `uploads/` 目录混用：`/api/raw` 高分辨率拍照、`/api/upload` 外部上传、`/api/photos/save` 仅打内存标签，三类文件堆在一起。"已保存到相册" 只是内存标记，文件物理位置不变。

- `album/` — 显式保存到相册的照片（从 `captures/` 移入）

`saved_photos` 条目新增 `location` 字段（值为 `'album'`），`url` 改为 `/album/<filename>`。新增 `/album/<filename>` 与 `/captures/<filename>` 路由服务静态文件。

```python
def migrate_legacy_uploads():
    """一次性迁移：将 saved_photos 中位于 uploads/ 的旧文件移到 album/。"""
    global saved_photos
    upload_dir = app.config['UPLOAD_FOLDER']
    album_dir = app.config['ALBUM_FOLDER']
    moved = 0
    for photo in saved_photos:
        fn = photo.get('filename')
        if not fn:
            continue
        # 已在 album/ 或 location 已是 album，跳过
        if photo.get('location') == 'album' and os.path.exists(os.path.join(album_dir, fn)):
            continue
        legacy = os.path.join(upload_dir, fn)
        if os.path.exists(legacy):
            import shutil
            shutil.move(legacy, os.path.join(album_dir, fn))
            photo['url'] = f'/album/{fn}'
            photo['location'] = 'album'
            moved += 1
    if moved:
        persist_state()
        print(f"迁移完成：{moved} 张旧相册照片移至 album/")
```

```cpp
// 上传图像到服务器
void uploadImage(camera_fb_t* fb, bool isHighResolution) {
  if (!fb) return;

  WiFiClient client;
  HTTPClient http;

  String url = String(SERVER_ADDRESS) + "/api/raw";
  if (http.begin(client, url)) {
    http.addHeader("Content-Type", "image/jpeg");
    http.addHeader("X-Resolution", isHighResolution ? "high" : "preview");
    http.addHeader("X-ESP32-IP", WiFi.localIP().toString());
    http.setTimeout(30000);

    Serial.printf("开始上传 %s 图像，大小: %zu 字节\n",
                  isHighResolution ? "高分辨率" : "预览", fb->len);

    unsigned long startTime = millis();
    int httpCode = http.POST((uint8_t*)fb->buf, fb->len);
    unsigned long uploadTime = millis() - startTime;

    if (httpCode > 0) {
      Serial.printf("上传成功，状态码: %d，耗时: %lu ms\n", httpCode, uploadTime);
    } else {
      Serial.printf("上传失败: %s，耗时: %lu ms\n",
                    http.errorToString(httpCode).c_str(), uploadTime);
    }
    http.end();
  } else {
    Serial.println("无法连接到服务器");
  }
}

// 执行一次拍照（高分辨率）并上传
void performCapture() {
  g_bIsCapturing = true;
  g_bPreviewRunning = false;

  esp_camera_deinit();
  delay(100);

  if (initCamera(true) == ESP_OK) {
    sensor_t *s = esp_camera_sensor_get();
    if (s) {
      s->set_awb_gain(s, 1);
      s->set_wb_mode(s, 0);
      s->set_aec2(s, 1);
      s->set_gain_ctrl(s, 1);
    }
    delay(500);

    if (g_bFlashEnabled) {
      digitalWrite(FLASH_GPIO_NUM, HIGH);
      delay(200);
    }

    // 丢弃前几帧让传感器适应
    for (int i = 0; i < 3; i++) {
      camera_fb_t *tmp = esp_camera_fb_get();
      if (tmp) esp_camera_fb_return(tmp);
      delay(100);
    }

    camera_fb_t *fb = esp_camera_fb_get();
    if (fb) {
      uploadImage(fb, true);
      esp_camera_fb_return(fb);
    } else {
      Serial.println("拍照失败");
    }

    if (g_bFlashEnabled) {
      digitalWrite(FLASH_GPIO_NUM, LOW);
    }
  }

  esp_camera_deinit();
  delay(100);
  initCamera(false);
  g_bIsCapturing = false;
  g_bPreviewRunning = true;
}

// 预览循环：持续上传低分辨率帧

    if (fb) {
      uploadImage(fb, false);
      esp_camera_fb_return(fb);
    }
  }
}

  if (request.indexOf("POST /capture") != -1) {
    g_bNeedCapture = true;
    client.println("HTTP/1.1 200 OK\r\nContent-Type: application/json\r\n\r\n{\"success\":true}");
  } else if (request.indexOf("POST /flash") != -1) {
    // 读取 body 判断 enable
    String body = client.readString();
    g_bFlashEnabled = body.indexOf("\"enable\":true") != -1;
    if (!g_bFlashEnabled) digitalWrite(FLASH_GPIO_NUM, LOW);
    client.println("HTTP/1.1 200 OK\r\nContent-Type: application/json\r\n\r\n{\"success\":true}");
  } else if (request.indexOf("GET /flash") != -1) {
    client.printf("HTTP/1.1 200 OK\r\nContent-Type: application/json\r\n\r\n{\"flash_enabled\":%s}",
                  g_bFlashEnabled ? "true" : "false");
  } else {
    client.println("HTTP/1.1 404 Not Found\r\n\r\n");
  }
  client.stop();
}
#endif

#ifdef ENABLE_FREERTOS_DUALCORE
// Core 0 任务：预览 + 拍照
void vTaskCore0(void *pvParameters) {
  Serial.println("Core 0 任务启动：预览与拍照");
  if (initCamera(false) != ESP_OK) {
    vTaskDelete(NULL);
    return;
  }
  g_bPreviewRunning = true;

  while (1) {
    if (g_bNeedCapture) {
      g_bNeedCapture = false;
      Serial.println("Core 0: 执行拍照");
      performCapture();
      continue;
    }
    previewLoop();
    vTaskDelay(pdMS_TO_TICKS(PREVIEW_INTERVAL_MS));
  }
}

#ifdef ENABLE_HTTP_CONTROL_SERVER
// Core 1 任务：HTTP 控制服务器
void vTaskCore1(void *pvParameters) {
  Serial.println("Core 1 任务启动：HTTP 控制服务器");
  WiFiServer server(80);
  server.begin();

  while (1) {
    WiFiClient client = server.available();
    if (client) {
      handleHttpControl(client);
    }
    vTaskDelay(pdMS_TO_TICKS(10));
  }
}
#endif
#endif

void setup() {
  Serial.begin(115200);
  Serial.println();
  Serial.println("ESP32-CAM 统一固件启动");

  pinMode(FLASH_GPIO_NUM, OUTPUT);
  digitalWrite(FLASH_GPIO_NUM, LOW);

  // 连接 WiFi
  Serial.printf("连接 WiFi: %s\n", WIFI_SSID);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println();
  Serial.print("IP 地址: ");
  Serial.println(WiFi.localIP());

#ifdef ENABLE_BLUETOOTH
  SerialBT.begin("esp32cam-unified");
  Serial.println("蓝牙已启动");
#endif

#ifdef ENABLE_FREERTOS_DUALCORE
  xTaskCreatePinnedToCore(vTaskCore0, "Core0Task", 8192, NULL, 1, &xTaskCore0, 0);
#ifdef ENABLE_HTTP_CONTROL_SERVER
  xTaskCreatePinnedToCore(vTaskCore1, "Core1Task", 4096, NULL, 1, &xTaskCore1, 1);
#endif
#else
  // 单核模式：直接初始化摄像头，loop() 中运行预览
  if (initCamera(false) != ESP_OK) {
    Serial.println("摄像头初始化失败，停止");
    while (1) delay(1000);
  }
  g_bPreviewRunning = true;
#endif
}

void loop() {
#ifdef ENABLE_FREERTOS_DUALCORE
  // 双核模式下 loop 空转，工作在任务中
  delay(1000);
#else
  // 单核模式
  if (g_bNeedCapture) {
    g_bNeedCapture = false;
    performCapture();
  } else {
    previewLoop();
    delay(PREVIEW_INTERVAL_MS);
  }

#ifdef ENABLE_HTTP_CONTROL_SERVER
  // 单核模式下在 loop 中轮询 HTTP（简化版）
  static WiFiServer server(80);
  static bool serverStarted = false;
  if (!serverStarted) {
    server.begin();
    serverStarted = true;
  }
  WiFiClient client = server.available();
  if (client) {
    handleHttpControl(client);
  }
#endif
#endif
}
```
